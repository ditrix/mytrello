### install 

composer create-project --prefer-dist laravel/laravel:^8.0 api-laravel

cd api-laravel

.env

DB_CONNECTION=mysql

DB_HOST=mysql

DB_PORT=3306

DB_DATABASE=api_laravel

DB_USERNAME=root

DB_PASSWORD=password


composer require laravel/sail --dev

php artisan sail:install
